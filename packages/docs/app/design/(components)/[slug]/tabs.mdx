### Default

<Editor
  scope={{ Tabs }}
  code={`() => {
  const [selected, setSelected] = React.useState('apple')
  
  return <Tabs
    tabs={[
      { title: 'Apple', value: 'apple' },
      { title: 'Orange', value: 'orange' },
      { title: 'Mango', value: 'mango' }
    ]}
    selected={selected}
    setSelected={setSelected}
  />
}`}
/>
