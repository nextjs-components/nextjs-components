### Sizes

<Editor
  scope={{ Combobox, Container }}
  code={`() => {
  const options = [
    { value: 'a', label: 'One' },
    { value: 'b', label: 'Two' },
    { value: 'c', label: 'Three' }
  ]
  
  return (
    <Container direction={['column', 'row']}>
      <Container>
        <Combobox placeholder="Search..." size="small">
          <Combobox.Input/>
          <Combobox.List>
            {options.map(option => (
              <Combobox.Option key={option.value} value={option.value}>
                {option.label}
              </Combobox.Option>
            ))}
          </Combobox.List>
        </Combobox>
      </Container>
  
      <Container>
        <Combobox placeholder="Search...">

### Prefix & Suffix

<Editor
  scope={{ Combobox, Facebook, Twitter, Instagram, Badge }}
  code={`() => {
  const options = [
    { value: 'a', label: 'Facebook', prefix: <Facebook size={16} /> },
    { value: 'b', label: 'Twitter', prefix: <Twitter size={16} /> },
    { value: 'c', label: 'Instagram', prefix: <Instagram size={16} /> }
  ]
  
  return (
    <Combobox placeholder="Search...">
      <Combobox.Input />
      <Combobox.List>
        {options.map(option => (
          <Combobox.Option
            key={option.value}
            value={option.value}
            prefix={option.prefix}
            suffix={
              <Badge size="small">
                {option.label}
              </Badge>
            }
          >
            {option.label}
          </Combobox.Option>
        ))}
      </Combobox.List>
    </Combobox>
  )
}`}
/>
