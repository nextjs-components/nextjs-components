# Toggle

<Spacer y={0.8} />

## Displays a boolean value.

<Spacer y={2} />

### Default

<Editor 
  scope={{Container, Toggle}}
  code={`
() => {
  const [checked, setChecked] = React.useState(false)
  const [checked2, setChecked2] = React.useState(true)

return (

<Container>
  <Container>
    <Toggle checked={checked} onChange={() => setChecked(!checked)} />
  </Container>
  <Container>
    <Toggle checked={checked2} onChange={() => setChecked2(!checked2)} />
  </Container>
</Container>
) }

`} />

<Spacer y={2} />
