# Switch

## Choose between a set of options.

<Spacer y={2} />

### Default

Ensure the width of each item is wide enough to prevent jumping when active.

<Spacer />

<Editor
  scope={{Container,Switch}}
  code={`() => {
  const [active, setActive] = React.useState('source')

return (

    <Container left>
      Active: {active}
      <Switch name="default" >
        <Switch.Control label="Source" defaultChecked value="source" onChange={e => setActive(e.target.value)}/>
        <Switch.Control label="Output" value="output" onChange={e => setActive(e.target.value)}/>
        <Switch.Control label="Other" value="other" onChange={e => setActive(e.target.value)}/>
      </Switch>
    </Container>

)
}`}
/>

### Disabled

<Editor
  scope={{Container,Switch}}
  code={`() => {
  const [active, setActive] = React.useState('source')

return (

    <Container left>
      <Switch
        items={[
          {
            name: 'Source',
            value: 'source',
            width: 80
          },
          {
            name: 'Output',
            value: 'output',
            width: 90,
            disabled: true
          }
        ]}
        active={active}
        onChange={v => setActive(v)}
      />
    </Container>

)
}`}
/>
