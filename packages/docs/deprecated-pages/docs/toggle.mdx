### Default

```
() => {
  const [checked, setChecked] = useState(false)
  const [checked2, setChecked2] = useState(true)

  return (
    <Container>
      <Container>
        <Toggle checked={checked} onChange={() => setChecked(!checked)} />
      </Container>

      <Container>
        <Toggle checked={checked2} onChange={() => setChecked2(!checked2)} />
      </Container>
    </Container>
  )
}
```
