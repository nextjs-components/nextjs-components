### Column Align

{/* prettier-ignore */}
<Text
  color="geist-secondary"
  size={16}
>Set the align property to change text alignment. Left by default.</Text>

<Spacer />

```
() => {
  const columns = useMemo(() => [
    {
      Header: 'First Name',
      accessor: 'first'
    },
    {
      Header: 'Last Name',
      accessor: 'last',
      align: 'center'
    },
    {
      Header: 'Email',
      accessor: 'email',
      align: 'right'
    }
  ], [])

  return (
    <Table
      columns={columns}
      data={data}
      caption="A table with right and left aligned columns."
    />
  )
}
```

### Column Class

{/* prettier-ignore */}
<Text
  color="geist-secondary"
  size={16}
>Passing columnClassName applies the class to all headers and cells of the column.</Text>

<Spacer />

```
() => {
  const columns = useMemo(() => [
    {
      Header: 'First Name',
      accessor: 'first'
    },
    {
      Header: 'Last Name',
      accessor: 'last',
      columnClassName: 'last-name-col'
    },
    {
      Header: 'Email',
      accessor: 'email'
    }
  ], [])

  return (
    <Table
      columns={columns}
      data={data}
      caption="A basic table with columnClassName applied to the second column."
    />
  )
}
```
