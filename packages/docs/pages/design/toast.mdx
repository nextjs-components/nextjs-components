import { DesignLayout, DesignLayoutInner } from "../../components/DesignLayout";

export default function Wrapper({ children, paths }) {
  return (
    <DesignLayout paths={paths}>
      <DesignLayoutInner>{children}</DesignLayoutInner>
    </DesignLayout>
  );
}

export const getStaticProps = async () => {
  const fs = await import("fs");
  const path = await import("path");
  const dir = path.join(process.cwd(), "pages/design");
  const paths = fs
    .readdirSync(dir)
    .map((e) => path.join("/design", e.replace(".mdx", "")));
  return {
    props: {
      paths,
    },
  };
};

# Toast

<Text h4 Component="h2" type="secondary" style={{ fontWeight: 400 }}>
  Display an important message globally.
</Text>

<Spacer y={2} />

### Default

```
() => {
  const toasts = useToasts()

  return (
    <Button
      onClick={() => {
        if (toasts && toasts.current) {
          toasts.current.message({
            text: 'The Netherland Dwarf jumped over the fence.'
          })
        }
      }}
    >
      Show Toast
    </Button>
  )
}
```

### Preserve

Preserved toasts will not automatically disappear.

```
() => {
  const toasts = useToasts()

  return (
    <Button
      onClick={() => {
        if (toasts && toasts.current) {
          toasts.current.message({
            text: 'The Netherland Dwarf jumped over the fence.',
            preserve: true
          })
        }
      }}
    >
      Show Toast
    </Button>
  )
}
```
